21bool UserDatabase::load(const std::string& db_path, Logger& logger) {
22 std::ifstream file(db_path);
23 if (!file.is_open()) {
24 logger.logError("Cannot open database file: " + db_path, true);
25 return false;
26 }
27
28 std::string line;
29 int line_number = 0;
30 while (std::getline(file, line)) {
31 line_number++;
32 if (line.empty() || line[0] == '#') {
33 continue;
34 }
35 size_t pos = line.find(':');
36 if (pos == std::string::npos) {
37 logger.logError("Invalid string format " + std::to_string(line_number), false);
38 continue;
39 }
40 std::string login = line.substr(0, pos);
41 std::string password = line.substr(pos + 1);
42 if (login.empty() || password.empty()) {
43 logger.logError("Empty login or password on line " + std::to_string(line_number), false);
44 continue;
45 }
46 users[login] = password;
47 }
48 file.close();
49
50 if (users.empty()) {
51 logger.logError("User database is empty", true);
52 return false;
53 }
54 logger.logInfo("Loaded " + std::to_string(users.size()) + " users from database");
55 return true;
56}
