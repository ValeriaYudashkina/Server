84std::string Server::readTextMessage(int sock) const {
85 char buffer[BUFLEN];
86 ssize_t rc = recv(sock, buffer, BUFLEN - 1, 0);
87 if (rc == -1) {
88 throw std::system_error(errno, std::generic_category(), "recv error reading MSG");
89 }
90 if (rc == 0) return "";
91
92 buffer[rc] = '\0';
93 std::string message(buffer);
94
95 message.erase(std::remove_if(message.begin(), message.end(),
96 [](char c){ return c == '\n' || c == '\r'; }), message.end());
97 return message;
98}

105void Server::startListening() {
106 listen_sock = socket(AF_INET, SOCK_STREAM, 0);
107 if (listen_sock == -1) {
108 throw std::system_error(errno, std::generic_category(), "socket creation failed");
109 }
110
111 int on = 1;
112 if (setsockopt(listen_sock, SOL_SOCKET, SO_REUSEADDR, (const char*)&on, sizeof on) == -1) {
113 logger.logError("setsockopt SO_REUSEADDR failed, continuing", false);
114 }
115
116 self_addr->sin_family = AF_INET;
117 self_addr->sin_port = htons(port);
118 self_addr->sin_addr.s_addr = INADDR_ANY;
119
120 if (bind(listen_sock, reinterpret_cast<const sockaddr*>(self_addr.get()), sizeof(sockaddr_in)) == -1) {
121 close(listen_sock);
122 throw std::system_error(errno, std::generic_category(), "bind failed");
123 }
124 if (listen(listen_sock, QLEN) == -1) {
125 close(listen_sock);
126 throw std::system_error(errno, std::generic_category(), "listen failed");
127 }
128 logger.logInfo("Server started and listening on port " + std::to_string(port));
129}

137void Server::run() {
138 startListening();
139 socklen_t socklen = sizeof(sockaddr_in);
140 while(true) {
141 int work_sock = -1;
142 try {
143 logger.logInfo("Waiting for new client...");
144 work_sock = accept(listen_sock, reinterpret_cast<sockaddr*>(foreign_addr.get()), &socklen);
145 if (work_sock == -1) {
146 logger.logError("Accept error: " + std::string(strerror(errno)), false);continue;
147 }
148 std::string ip_addr(inet_ntoa(foreign_addr->sin_addr));
149 logger.logInfo("Connection established with " + ip_addr);
150 handleClient(work_sock);
151 } catch (const std::exception& e) {
152 logger.logError("Error in server loop: " + std::string(e.what()), false);
153 }
154 if (work_sock != -1) {
155 close(work_sock);
156 logger.logInfo("Connection closed");
157 }
158 }
159}

171void Server::handleClient(int client_sock) {
172 try {
173 std::string full_msg = readTextMessage(client_sock);
174 if (full_msg.empty()) {
175 logger.logError("Client disconnected during authentication", false);
176 return;
177 }
178 if (full_msg.length() < AUTH_DATA_LENGTH) {
179 throw auth_error("Auth message too short");
180 }
181 std::string auth_data = full_msg.substr(full_msg.length() - AUTH_DATA_LENGTH);
182 std::string login = full_msg.substr(0, full_msg.length() - AUTH_DATA_LENGTH);
183
184 if (authenticator.verify(login, auth_data, userDb, logger)) {
185 const char* ok_msg = "OK";
186 send(client_sock, ok_msg, strlen(ok_msg), 0);
187 logger.logInfo("Client '" + login + "' authenticated successfully");
188 } else {
189 throw auth_error("Authentication failed for login " + login);
190 }
191 processVectors(client_sock);
192 } catch (const auth_error& e) {
193 sendError(client_sock, e.what());
194 throw;
195 } catch (const std::exception& e) {
196 sendError(client_sock, "Protocol processing error");
197 throw;
198 }
199}

214void Server::processVectors(int sock) {
215 uint32_t num_vectors;
216 ssize_t rc;
217
218 rc = recv(sock, &num_vectors, sizeof(num_vectors), 0);
219 if (rc != sizeof(num_vectors)) {
220 throw vector_error("Failed to receive number of vectors");
221 }
222 logger.logInfo("Receiving " + std::to_string(num_vectors) + " vectors");
223 for (uint32_t i = 0; i < num_vectors; ++i) {
224 uint32_t vector_len;
225 rc = recv(sock, &vector_len, sizeof(vector_len), 0);
226 if (rc != sizeof(vector_len)) {
227 throw vector_error("Failed to receive vector length");
228 }
229
230 size_t total_bytes_needed = vector_len * sizeof(int32_t);
231
232 if (vector_len == 0 || total_bytes_needed > 4000000000) {
233 throw vector_error("Vector size invalid or too large");
234 }
235 std::vector<int32_t> data(vector_len);
236 rc = recv(sock, data.data(), total_bytes_needed, 0);
237 if (rc != (ssize_t)total_bytes_needed) {
238 throw vector_error("Vector data size mismatch");
239 }
240 int32_t result = processor.calculateAverage(data, logger);
241 int32_t net_result = (result);
242 send(sock, &net_result, sizeof(net_result), 0);
243
244 logger.logInfo("Processed vector " + std::to_string(i+1) + ", result: " + std::to_string(result));
245 }
